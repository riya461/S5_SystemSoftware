//Subtract two 16-bit numbers

ASSUME CS:CODE, DS:DATA

DATA SEGMENT
    NUM1 DW 5678H
    NUM2 DW 1234H
    RESULT DW 01 DUP(?)
DATA ENDS

CODE SEGMENT
START:
    MOV AX, DATA
    MOV DS, AX
    XOR AX, AX
    MOV AX, NUM1
    SUB AX, NUM2
    MOV RESULT, AX
    MOV AH, 4CH
    INT 21H
CODE ENDS

END START
